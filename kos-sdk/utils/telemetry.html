 62    async def start(self) -> None:
 63        """Start telemetry logging."""
 64        if self._is_logging:
 65            return
 66
 67        timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
 68
 69        # Open log files and initialize CSV writers
 70
 71        self.imu_file = open(self.log_dir / f"imu_{timestamp}.csv", "w", newline="")
 72        self.actuator_file = open(self.log_dir / f"actuator_{timestamp}.csv", "w", newline="")
 73        self.control_file = open(self.log_dir / f"control_{timestamp}.csv", "w", newline="")

 81                    "roll",
 82                    "pitch",

 84                    "accel_x",
 85                    "accel_y",
 86                    "accel_z",  # Linear acceleration
 87                    "gyro_x",
 88                    "gyro_y",
 89                    "gyro_z",  # Angular velocity
 90                    "quat_w",
 91                    "quat_x",
 92                    "quat_y",
 93                    "quat_z",  # Quaternion
 94                ]
 95            )
 96
 97        if self.actuator_file:
 98            self.actuator_writer = csv.writer(self.actuator_file)
 99            self.actuator_writer.writerow(
100                [
101                    "timestamp",
102                    "actuator_id",
103                    "position",
104                    "velocity_rad_s",
105                    "torque",
106                    "current",
107                    "temperature",
108                    "voltage",
109                    "online",
110                    "faults",

122                ]

126            self._is_logging = True
127
128        # Start background logging task

132        """Stop telemetry logging and close files."""

141    async def _log_loop(self) -> None:
142        """Background task for continuous logging."""
143        while self._is_logging:
144            try:
145                await self._log_single_frame()
146            except Exception as e:
147                logger.error(f"Error in telemetry logging: {e}")
148            await asyncio.sleep(0.01)  # 100Hz target rate

150    async def _log_single_frame(self) -> None:
151        """Log a single frame of telemetry data."""
152        timestamp = datetime.datetime.now().isoformat()
153
154        # Log IMU data
155        try:
156            cmd_start = time.time()
157            imu_euler = await self.kos.imu.get_euler_angles()
158            imu_values = await self.kos.imu.get_imu_values()
159            quat = await self.kos.imu.get_quaternion()
160            cmd_latency = time.time() - cmd_start
161
162            if self.imu_writer is not None:
163                self.imu_writer.writerow(
164                    [
165                        timestamp,
166                        imu_euler.roll,
167                        imu_euler.pitch,
168                        imu_euler.yaw,
169                        imu_values.accel_x,
170                        imu_values.accel_y,
171                        imu_values.accel_z,
172                        imu_values.gyro_x,
173                        imu_values.gyro_y,
174                        imu_values.gyro_z,
175                        quat.w,
176                        quat.x,
177                        quat.y,
178                        quat.z,
179                    ]
180                )
181        except Exception as e:
182            logger.warning(f"Failed to get IMU data: {e}")
183
184        # Log actuator data
185        try:
186            cmd_start = time.time()
187            states = await self.kos.actuator.get_actuators_state(self.actuator_ids)
188            cmd_latency = time.time() - cmd_start
189
190            for actuator_id, state in zip(self.actuator_ids, states.states):
191                if self.actuator_writer is not None:
192                    self.actuator_writer.writerow(
193                        [
194                            timestamp,
195                            actuator_id,
196                            state.position,
197                            state.velocity,
198                            state.torque,
199                            state.current,
200                            state.temperature,
201                            state.voltage,
202                            state.online,
203                            ",".join(state.faults) if state.faults else "",
204                        ]
205                    )
206        except Exception as e:
207            logger.warning(f"Failed to get actuator data: {e}")
208
209        # Calculate and log control metrics
210        current_time = time.time()
211        loop_duration = current_time - self.last_loop_time
212        self.last_loop_time = current_time
213        self.loop_times.append(loop_duration)
214
215        if len(self.loop_times) > 100:  # Keep a rolling window
216            self.loop_times.pop(0)
217
218        avg_frequency = 1.0 / (sum(self.loop_times) / len(self.loop_times))
219
220        if self.control_writer is not None:
221            self.control_writer.writerow(
222                [
223                    timestamp,
224                    avg_frequency,
225                    cmd_latency,
226                    cmd_latency,
227                ]  # Using command latency as proxy for gRPC latency
228            )

231def plot_latest_logs(log_dir: str = "telemetry_logs") -> None:
232    """Plot the most recent telemetry logs."""
233    # Find latest log files
234    imu_files = sorted(Path(log_dir).glob("imu_*.csv"))
235    actuator_files = sorted(Path(log_dir).glob("actuator_*.csv"))
236    control_files = sorted(Path(log_dir).glob("control_*.csv"))
237
238    if not (imu_files and actuator_files and control_files):
239        raise FileNotFoundError("No log files found in directory")
240
241    # Load most recent files
242    imu_data = pd.read_csv(imu_files[-1])
243    actuator_data = pd.read_csv(actuator_files[-1])
244    control_data = pd.read_csv(control_files[-1])
245
246    # Convert timestamps to relative seconds for plotting
247    for df in [imu_data, actuator_data, control_data]:
248        df["timestamp"] = pd.to_datetime(df["timestamp"])
249        df["time"] = (df["timestamp"] - df["timestamp"].min()).dt.total_seconds()
250
251    # Create plots using code from plot.py
252    plt.style.use("ggplot")
253    fig = plt.figure(figsize=(15, 24))
254    gs = GridSpec(8, 2, figure=fig)
255
256    # IMU Data Plots (Left Column)
257    plot_imu_data(fig, gs, imu_data)
258
259    # Control Metrics (Bottom Left)
260    plot_control_metrics(fig, gs, control_data)
261
262    # Actuator Data (Right Column)
263    plot_actuator_data(fig, gs, actuator_data)
264
265    plt.tight_layout()
266
267    # Save plot
268    timestamp = pd.Timestamp.now().strftime("%Y%m%d_%H%M%S")
269    plot_filename = f"telemetry_plot_{timestamp}.png"
270    plt.savefig(plot_filename, dpi=300, bbox_inches="tight")
271    logger.info(f"Plot saved as {plot_filename}")
272
273    plt.show()

276def plot_imu_data(fig: plt.Figure, gs: GridSpec, imu_data: pd.DataFrame) -> None:
277    """Plot IMU data in the left column."""
278    # Euler Angles
279    ax_euler = fig.add_subplot(gs[0, 0])
280    ax_euler.plot(imu_data["time"], imu_data["roll"], label="Roll")
281    ax_euler.plot(imu_data["time"], imu_data["pitch"], label="Pitch")
282    ax_euler.plot(imu_data["time"], imu_data["yaw"], label="Yaw")
283    ax_euler.set_title("IMU Orientation (Euler Angles)")
284    ax_euler.set_xlabel("Time (s)")
285    ax_euler.set_ylabel("Angle (°)")
286    ax_euler.grid(True)
287    ax_euler.legend()
288
289    # Acceleration
290    ax_accel = fig.add_subplot(gs[1, 0])
291    ax_accel.plot(imu_data["time"], imu_data["accel_x"], label="X")
292    ax_accel.plot(imu_data["time"], imu_data["accel_y"], label="Y")
293    ax_accel.plot(imu_data["time"], imu_data["accel_z"], label="Z")
294    ax_accel.set_title("Linear Acceleration")
295    ax_accel.set_xlabel("Time (s)")
296    ax_accel.set_ylabel("Acceleration (m/s²)")
297    ax_accel.grid(True)
298    ax_accel.legend()
299
300    # Angular Velocity
301    ax_gyro = fig.add_subplot(gs[2, 0])
302    ax_gyro.plot(imu_data["time"], imu_data["gyro_x"], label="X")
303    ax_gyro.plot(imu_data["time"], imu_data["gyro_y"], label="Y")
304    ax_gyro.plot(imu_data["time"], imu_data["gyro_z"], label="Z")
305    ax_gyro.set_title("Angular Velocity")
306    ax_gyro.set_xlabel("Time (s)")
307    ax_gyro.set_ylabel("Angular Velocity (rad/s)")
308    ax_gyro.grid(True)
309    ax_gyro.legend()

312def plot_control_metrics(fig: plt.Figure, gs: GridSpec, control_data: pd.DataFrame) -> None:
313    """Plot control metrics."""
314    ax_control = fig.add_subplot(gs[4, 0])
315    ax_control.plot(control_data["time"], control_data["loop_frequency"], label="Loop Frequency")
316    ax_control2 = ax_control.twinx()
317    ax_control2.plot(
318        control_data["time"], control_data["command_latency"] * 1000, "r-", label="Command Latency"
319    )
320    ax_control.set_title("Control Performance Metrics")
321    ax_control.set_xlabel("Time (s)")
322    ax_control.set_ylabel("Frequency (Hz)")
323    ax_control2.set_ylabel("Latency (ms)")
324    ax_control.grid(True)
325    ax_control.legend(loc="upper left")
326    ax_control2.legend(loc="upper right")

329def plot_actuator_data(fig: plt.Figure, gs: GridSpec, actuator_data: pd.DataFrame) -> None:
330    """Plot actuator data in the right column."""
331    axes = {
332        "position": fig.add_subplot(gs[0, 1]),
333        "velocity_rad_s": fig.add_subplot(gs[1, 1]),
334        "torque": fig.add_subplot(gs[2, 1]),
335        "current": fig.add_subplot(gs[3, 1]),
336        "temperature": fig.add_subplot(gs[4, 1]),
337        "voltage": fig.add_subplot(gs[5, 1]),
338    }
339
340    titles = {
341        "position": ("Actuator Positions", "Position (rad)"),
342        "velocity_rad_s": ("Actuator Velocities", "Velocity (rad/s)"),
343        "torque": ("Actuator Torques", "Torque (Nm)"),
344        "current": ("Actuator Currents", "Current (A)"),
345        "temperature": ("Actuator Temperatures", "Temperature (°C)"),
346        "voltage": ("Actuator Voltages", "Voltage (V)"),
347    }
348
349    for actuator_id in actuator_data["actuator_id"].unique():
350        mask = actuator_data["actuator_id"] == actuator_id
351        data = actuator_data[mask]
352
353        for param, ax in axes.items():
354            ax.plot(data["time"], data[param], label=f"ID {actuator_id}")
355
356    for param, (title, ylabel) in titles.items():
357        ax = axes[param]
358        ax.set_title(title)
359        ax.set_xlabel("Time (s)")
360        ax.set_ylabel(ylabel)
361        ax.grid(True)
362        ax.legend()
